  // ========================== HELPERS (oryginalne) ==========================
  function estimateTokens(text) { return Math.round(text.length / 4); }

  function fixNewlinesInJsonStrings(str) {
    let result = '', inString = false, escaped = false;
    for (let i = 0; i < str.length; i++) {
      const ch = str[i];
      if (escaped) { result += ch; escaped = false; continue; }
      if (ch == '\\' && inString) { result += ch; escaped = true; continue; }
      if (ch == '"') { inString = !inString; result += ch; continue; }
      if (inString) {
        if (ch == '\n') { result += '\\n'; continue; }
        if (ch == '\r') { result += '\\r'; continue; }
        if (ch == '\t') { result += '\\t'; continue; }
        const code = ch.charCodeAt(0);
        if (code < 0x20) { result += '\\u' + code.toString(16).padStart(4, '0'); continue; }
      }
      result += ch;
    }
    return result;
  }

  function autoFix(raw) {
    let text = raw.trim();
    text = text.replace(/^\uFEFF/, '').replace(/\0/g, '');
    text = text.replace(/^```(?:\w+)?\s*/i, '').replace(/\s*```\s*$/, '').trim();
    const f = text.indexOf('{'), l = text.lastIndexOf('}');
    if (f != -1 && l != -1 && l > f && (f > 0 || l < text.length - 1))
      text = text.slice(f, l + 1);
    text = text.replace(/,\s*([\]}])/g, '$1');
    text = fixNewlinesInJsonStrings(text);
    text = text.replace(/(?<!\\)\\(?![\\/bfnrtu])/g, '\\\\');
    return text;
  }

  function checkThinkDensity(thinkContent, issues) {
    const sentences = thinkContent.split(/[.!?;]\s+/).filter(s => s.trim().length > 0);
    if (sentences.length < 5) return;
    const ngramCounts = new Map();
    for (const sentence of sentences) {
      const words = sentence.toLowerCase().split(/\s+/).filter(w => w.length > 2);
      if (words.length < 5) continue;
      for (let i = 0; i <= words.length - 5; i++) {
        const ngram = words.slice(i, i + 5).join(' ');
        ngramCounts.set(ngram, (ngramCounts.get(ngram) || 0) + 1);
      }
      if (words.length >= 5) {
        const fullNgram = words.join(' ');
        ngramCounts.set(fullNgram, (ngramCounts.get(fullNgram) || 0) + 1);
      }
    }
    const fluffAlerts = [];
    for (const [ngram, count] of ngramCounts.entries()) {
      if (count >= 3) fluffAlerts.push({ ngram, count });
    }
    if (fluffAlerts.length > 0) {
      const topFluff = fluffAlerts.sort((a, b) => b.count - a.count).slice(0, 3);
      for (const fluff of topFluff) {
        issues.push('ANTI-FLUFF v69.0: N-gram powtórzony ' + fluff.count + 'x w <think>: "' + fluff.ngram.substring(0, 80) + '..." - myślenie musi być gęste, bez lania wody!');
      }
    }
  }

      const importPatterns = content.match(/\b(import\s+\w+|from\s+\w+\s+import|require\s*\(|use\s+\w+::|using\s+\w+\.|#include\s*<)/gi);
      if (importPatterns && importPatterns.length > 3) {
        issues.push('STRICT_CONSTRAINT_VIOLATION v69.0: Kod nie może używać zewnętrznych bibliotek ale zawiera liczne importy');
      }
    }
    if (hasMaxLines) {
      const maxLines = parseInt(hasMaxLines[1]);
      const afterThink = content.indexOf('</think>');
      if (afterThink > -1) {
        const codePart = content.substring(afterThink + 8);
        const codeLines = codePart.split('\\n').filter(l => l.trim().length > 0 && !l.trim().startsWith('//')).length;
        if (codeLines > maxLines) {
          issues.push('STRICT_CONSTRAINT_VIOLATION v69.0: Kod ma ' + codeLines + ' linii, limit to ' + maxLines);
        }
      }
    }
  }

  // ========================== v69.0: SYSTEM FAILURE VALIDATION ==========================
  function validateSystemFailures(content, issues, warnings) {
    const systemFailurePatterns = [
      { pattern: /segmentation\s*fault/i, name: 'Segmentation fault' },
      { pattern: /out\s*of\s*memory/i, name: 'Out of Memory' },
      { pattern: /connection\s*timed?\s*out/i, name: 'Connection timed out' },
      { pattern: /disk\s*quota\s*exceeded/i, name: 'Disk quota exceeded' },
      { pattern: /too\s*many\s*open\s*files/i, name: 'Too many open files' },
      { pattern: /EAGAIN/i, name: 'EAGAIN' },
    ];
    let foundFailures = [];
    for (const fp of systemFailurePatterns) {
      if (fp.pattern.test(content)) foundFailures.push(fp.name);
    }
    if (foundFailures.length > 0) {
      const hasSysadminResponse = /alokacj[ęi]|allocation|retry|backoff|timeout.*handling|restrukturyzacj[ęi]|sysctl|ulimit|OOM.*score|memory.*pressure|buffer.*size|chunk.*size|mmap|madvise|mprotect|resource.*limit/i;
      const hasSystemDiagnosis = /diagnoz[ae]|kernel|proces|signal|dmesg|journalctl|syslog|kill\s*-\d+|SIGKILL|SIGSEGV|oom_reaper|oom_score_adj/i;
      if (!hasSysadminResponse.test(content)) {
        issues.push('SYSTEM_FAILURE_VALIDATION v69.0: Wykryto błąd systemowy (' + foundFailures.join(', ') + ') ale agent NIE reaguje jak sysadmin – brak retry/backoff/zmiany alokacji!');
      } else {
        warnings.push('✅ SYSTEM_FAILURE v69.0: Agent poprawnie reaguje na błąd systemowy (' + foundFailures.join(', ') + ')');
      }
      if (!hasSystemDiagnosis.test(content)) {
        warnings.push('⚠ SYSTEM_FAILURE v69.0: Agent reaguje na błąd systemowy ale brak diagnozy na poziomie OS (kernel/procesy/signal)');
      }
      const hasRealisticFormat = /core\s*dumped|0x[0-9a-f]+|killed\s*process|exit\s*code|signal\s*\d+/i;
      if (!hasRealisticFormat.test(content)) {
        warnings.push('⚠ SYSTEM_FAILURE v69.0: Błąd systemowy ma nierealistyczny format – dodaj adresy pamięci, kody wyjścia, nazwy sygnałów');
      }
    }
  }

      'C++': ['iostream', 'fstream', 'sstream', 'string', 'vector', 'map', 'set', 'algorithm', 'memory', 'thread', 'mutex', 'chrono', 'filesystem', 'regex', 'random']
    };
    const allowedModules = stdlibModules[language] || [];
    if (allowedModules.length === 0) return;
    const importRegex = /import\s+(\w[\w.]*)/g;
    let match;
    while ((match = importRegex.exec(content)) !== null) {
      const mod = match[1];
      if (!allowedModules.some(allowed => mod.startsWith(allowed))) {
        issues.push('FAKE_API v70.0: Podejrzany import "' + mod + '" – nie znajduje się w standardowej bibliotece ' + language + ' i może być halucynacją.');

  // 2. Cross-File Consistency
  function checkCrossFileConsistency(content, issues) {
    const fileSections = content.split(/\/\/ FILE:\s*(.+)/);
    const files = {};
    for (let i = 1; i < fileSections.length; i += 2) {
      const filePath = fileSections[i].trim();
      const fileContent = fileSections[i + 1] || '';
      files[filePath] = fileContent;
    }
    const definedSymbols = {};
    for (const [path, code] of Object.entries(files)) {
      const defs = [];
      const funcRegex = /(?:function|def|func|fn|class|const|let|var)\s+(\w+)/g;
      let m;
      while ((m = funcRegex.exec(code)) !== null) defs.push(m[1]);
      definedSymbols[path] = defs;
    }
    for (const [path, code] of Object.entries(files)) {
      const importRegex = /(?:import|require|from)\s+(?:\{?\s*(\w+)\s*\}?\s*from\s+['"]\.?\/?([^'"]+)['"]|['"](\w+)['"])/g;
      let m;
      while ((m = importRegex.exec(code)) !== null) {
        const importedName = m[1] || m[3];
        const targetFile = m[2];
        if (!importedName) continue;
        let found = false;
        for (const [fpath, symbols] of Object.entries(definedSymbols)) {
          if (fpath.includes(targetFile)) {
            if (symbols.includes(importedName)) found = true;
            break;
          }
        }
        if (!found) {
          issues.push('CROSS_FILE v70.0: Import "' + importedName + '" w pliku ' + path + ' nie został znaleziony w pliku docelowym.');
          stats.crossFileInconsistencies++;
        }
      }
    }
  }

  // 3. Dead Code Detection
  function detectDeadCode(content, issues) {
    const funcDefs = content.match(/(?:function|def|fn|func|class|const|let|var)\s+(\w+)/g) || [];
    const usedSymbols = new Set();
    const usageRegex = /\b(\w+)\s*\(/g;
    let m;
    while ((m = usageRegex.exec(content)) !== null) usedSymbols.add(m[1]);
    for (const def of funcDefs) {
      const name = def.split(/\s+/)[1];
      if (!usedSymbols.has(name) && !['main', 'index', 'handler', 'init'].includes(name)) {
        issues.push('DEAD_CODE v70.0: Zdefiniowano "' + def + '" ale nigdzie nie użyto.');
        stats.deadCodeOccurrences++;
      }
    }
  }

  // 4. Semantic Duplicate Detection
  const recentFingerprints = [];
  function detectSemanticDuplicate(parsed) {
    if (!parsed.metadata) return false;
    const fingerprint = `${parsed.metadata.language}|${parsed.metadata.domain}|${parsed.metadata.project_type}|${parsed.metadata.unique_twist}`;
    if (recentFingerprints.includes(fingerprint)) {
      stats.semanticDuplicates++;
      return true;
    }
    recentFingerprints.push(fingerprint);
    if (recentFingerprints.length > 20) recentFingerprints.shift();
    return false;
  }

  // 5. Cyclomatic Complexity Gate
  function checkCyclomaticComplexity(content, difficulty, issues) {
    const complexity = (content.match(/\b(if|else\s+if|for|while|switch|case\b|catch)\b/g) || []).length + 1;
    const limits = { easy: [1, 4], medium: [5, 12], hard: [13, 25], extreme: [26, 100] };
    const [min, max] = limits[difficulty] || [0, Infinity];
    if (complexity < min || complexity > max) {
      issues.push('COMPLEXITY v70.0: Złożoność cyklomatyczna ' + complexity + ' poza zakresem dla ' + difficulty + ' (' + min + '-' + max + ').');
      stats.complexityViolations++;
    }
  }

  // 6. Dependency Version Consistency
  function checkDependencyVersions(content, issues) {
    const versionRegex = /"version"\s*:\s*"(\d+\.\d+\.\d+)"/g;
    let m;
    while ((m = versionRegex.exec(content)) !== null) {
      const ver = m[1];
      if (!/^\d+\.\d+\.\d+$/.test(ver)) {
        issues.push('DEP_VERSION v70.0: Niewłaściwy format wersji "' + ver + '".');
        stats.dependencyVersionIssues++;
      }
    }
  }

  // 7. Token Efficiency Score
  function checkTokenEfficiency(content, issues) {
    const operationalCount = (content.match(/(<tool_call|app\.(get|post|put|delete)|db\.(query|execute)|\.test\(|\.assert)/g) || []).length;
    const tokens = estimateTokens(content);
    if (tokens > 0 && operationalCount / tokens < 0.02) {
      issues.push('TOKEN_EFF v70.0: Niska gęstość operacji (' + (operationalCount/tokens*100).toFixed(1) + '%) – kod może być rozwlekły.');
      stats.tokenEfficiencyWarnings++;
    }
  }

  // 8. Adversarial Self-Test
  function checkAdversarialTests(content, issues) {
    const testSection = content.match(/(?:describe|it|test|def test_|fn test_)/g) || [];
    const adversarialIndicators = /(?:empty|null|undefined|NaN|extreme|boundary|invalid|negative|zero|large|overflow|injection|escape|script)/i;
    let hasAdversarial = false;
    for (const test of testSection) {
      if (adversarialIndicators.test(test)) { hasAdversarial = true; break; }
    }
    if (!hasAdversarial && testSection.length > 0) {
      issues.push('ADVERSARIAL v70.0: Brak testów przypadków brzegowych (np. puste wejście, ekstremalne wartości).');
      stats.adversarialTestMissing++;
    }
  }

  // ========================== GŁÓWNA WALIDACJA STRUKTURY (rozszerzona) ==========================
  function validateStructure(parsed) {
    const issues = [], warnings = [];
    const difficulty = parsed?.metadata?.difficulty || 'unknown';
    if (!parsed.messages || !Array.isArray(parsed.messages)) {
      issues.push('Brak pola "messages"');
      return { valid: false, issues, warnings };
    }
    if (parsed.messages.length != 3)
      issues.push('messages ma ' + parsed.messages.length + ' el. zamiast 3');
    ['system', 'user', 'assistant'].forEach((role, i) => {
      if (parsed.messages[i]?.role != role)
        issues.push('messages[' + i + '].role != "' + role + '"');
      if (!parsed.messages[i]?.content || parsed.messages[i].content.length < 10)
        issues.push('messages[' + i + '].content za krótki');
    });
    const am = parsed.messages.find(m => m.role === 'assistant');
    if (am?.content) {
      const c = am.content;
      if (!c.includes('<think>')) issues.push('Brak <think>');
      if (!c.includes('</think>')) issues.push('Brak </think>');
      const ts = c.indexOf('<think>');
      const te = c.indexOf('</think>');
      if (ts != -1 && te != -1 && te > ts) {
        const thinkContent = c.slice(ts + 7, te);
        const thinkTokens = estimateTokens(thinkContent);
        if (thinkTokens < 400) issues.push('<think> za krótki: ~' + thinkTokens + ' tokenów (min 400)');
        if (!thinkContent.includes('## 1. Architektura i Ograniczenia'))
          issues.push('<think> brak sekcji "## 1. Architektura i Ograniczenia" (Priorytet 5)');
        if (!thinkContent.includes('## 2. Threat Modeling'))
          issues.push('<think> brak sekcji "## 2. Threat Modeling" (Priorytet 5)');
        if (!thinkContent.includes('## 3. Strategia ReAct'))
          issues.push('<think> brak sekcji "## 3. Strategia ReAct" (Priorytet 5)');
        if (!thinkContent.includes('## 4. JSONL Self-Check'))
          issues.push('<think> brak sekcji "## 4. JSONL Self-Check" (Priorytet 5)');
        if (!thinkContent.includes('PASS') && !thinkContent.includes('VALIDATION') && !thinkContent.includes('CRITIQUE') && !thinkContent.includes('REPAIR')) {
          warnings.push('PRIORYTET 6: Brak śladów Multi-Pass Review (PASS 1/2/3) w sekcji <think> - reviewer powinien dokumentować kroki weryfikacji');
        }
        checkThinkDensity(thinkContent, issues);
      }
      if (te != -1) {
        const after = c.slice(te + 8).trim();
        if (after.length < 150) issues.push('Output po </think> za krótki');
        if (!(after.match(/\/\/ FILE:/g) || []).length) issues.push('Brak // FILE:');
        if (!after.includes('<tool_call')) issues.push('Brak <tool_call>');
        const afterTokens = estimateTokens(after);
        const minTokens = { easy: 150, medium: 400, hard: 800, extreme: 1400 };
        if (minTokens[difficulty] && afterTokens < minTokens[difficulty])
          issues.push('Output za krótki dla ' + difficulty + ': ~' + afterTokens + ' tokenów (min ' + minTokens[difficulty] + ')');
        if (!after.includes('git_commit')) {
          if (difficulty == 'easy') warnings.push('Brak git_commit (opcjonalny easy)');
          else if (difficulty != 'unknown') issues.push('Brak git_commit (wymagany medium+)');
          else warnings.push('Brak git_commit');
        }
        if (!after.includes('run_tests')) {
          if (difficulty === 'easy') warnings.push('Brak run_tests (opcjonalne easy)');
          else if (difficulty !== 'unknown') issues.push('Brak run_tests (wymagane medium+)');
          else warnings.push('Brak run_tests');
        }
        if (!after.includes('run_security_scan') && (difficulty === 'hard' || difficulty === 'extreme'))
          issues.push('Brak run_security_scan (wymagane hard/extreme)');
        if (!after.includes('BENCHMARK') && (difficulty === 'hard' || difficulty === 'extreme'))
          warnings.push('Brak BENCHMARK (zalecane hard/extreme)');
        if (!after.includes('retry') && !after.includes('fallback') && !after.includes('circuit') && !after.includes('SELF-CORRECTION'))
          warnings.push('Brak error recovery pattern');
        if (!after.includes('zod') && !after.includes('pydantic') && !after.includes('class-validator') && !after.includes('validate'))
          warnings.push('Brak walidacji inputów');
        validateToolCalls(after, issues);
        validateCodeQuality(after, issues, warnings);

        // v69.0 checks
        verifyStrictConstraints(c, issues);
        validateSystemFailures(c, issues, warnings);

        // v70.0 NOWE CHECKS
        detectFakeApiCalls(after, parsed.metadata?.language || 'unknown', issues);
        checkCrossFileConsistency(after, issues);
        detectDeadCode(after, issues);
        checkCyclomaticComplexity(after, difficulty, issues);
        checkDependencyVersions(after, issues);
        checkTokenEfficiency(after, issues);
        checkAdversarialTests(after, issues);

        if (detectSemanticDuplicate(parsed)) {
          issues.push('SEMANTIC_DUPLICATE v70.0: Przykład ma identyczny fingerprint jak jeden z ostatnich 20.');
        }

        // chaos/deep_repo checks
        if (parsed.metadata?.is_chaotic_user === true) {
          const firstWriteIdx = c.indexOf('<tool_call name="write_file"');
          const firstReplaceIdx = c.indexOf('<tool_call name="replace_in_file"');
          const firstDeleteIdx = c.indexOf('<tool_call name="delete_file"');
          const firstMod = Math.min(
            firstWriteIdx > -1 ? firstWriteIdx : Infinity,
            firstReplaceIdx > -1 ? firstReplaceIdx : Infinity,
            firstDeleteIdx > -1 ? firstDeleteIdx : Infinity
          );
          const askIdx = c.indexOf('<tool_call name="ask_clarification"');
          if (firstMod < Infinity && askIdx === -1) {
            issues.push('PRIORYTET 6: is_chaotic_user:true – brak ask_clarification przed modyfikacjami plików!');
          } else if (firstMod < Infinity && askIdx > firstMod) {
            issues.push('PRIORYTET 6: is_chaotic_user:true – ask_clarification użyte PO modyfikacjach plików!');
          }
        }
        if (parsed.metadata?.is_deep_repo === true) {
          if (!c.includes('<tool_call name="list_files"')) {
            issues.push('PRIORYTET 6: is_deep_repo:true – brak list_files!');
          }
          if (!c.includes('<tool_call name="search_code"')) {
            issues.push('PRIORYTET 6: is_deep_repo:true – brak search_code!');
          }
          if (!c.includes('<tool_call name="read_file"')) {
            issues.push('PRIORYTET 6: is_deep_repo:true – brak read_file!');
          }
        }
      }
      if (c.includes('---TURN_BREAK---')) {
        const turns = c.split('---TURN_BREAK---');
        turns.forEach((t, i) => {
          if (!t.includes('<think>')) issues.push('Tura ' + (i + 1) + ' bez <think>');
          if (!t.includes('</think>')) issues.push('Tura ' + (i + 1) + ' bez </think>');
        });
        if ((difficulty === 'hard' || difficulty === 'extreme') && parsed.metadata?.is_multi_turn === true) {
          const toolCallMatches = c.match(/<tool_call\s+name="/g) || [];
          if (toolCallMatches.length < 3) {
            issues.push('PRIORYTET 6: hard/extreme multi-turn wymaga minimum 3 unikalnych interakcji z narzędziami, wykryto: ' + toolCallMatches.length);
          }
        }
      }
    }
    if (parsed.metadata?.is_multi_turn === true && !c.includes('---TURN_BREAK---'))
      issues.push('is_multi_turn:true ale brak TURN_BREAK');
    if (parsed.metadata?.is_multi_turn === false && c.includes('---TURN_BREAK---'))
      issues.push('is_multi_turn:false ale jest TURN_BREAK');
    if (parsed.metadata?.is_negative_example === true) {
      if (!c.includes('SELF-CORRECTION') && !c.includes('INCORRECT') && !c.includes('błąd') && !c.includes('error') && !c.includes('napraw')) {
        warnings.push('PRIORYTET 6: is_negative_example:true ale brak śladów błędu i jego naprawy w treści - przykład negatywny musi demonstrować wykrycie i korekcję błędu');
      }
    }
    const bannedTwistKeywords = ['Auth', 'JWT', 'CRUD', 'Todo', 'UserManagement', 'Blogging', 'ShoppingCart', 'PaymentGateway'];
    if (parsed.metadata?.unique_twist) {
      for (const kw of bannedTwistKeywords) {
        if (parsed.metadata.unique_twist.toLowerCase().includes(kw.toLowerCase())) {
          warnings.push('PRIORYTET 6: unique_twist zawiera banalną domenę "' + kw + '" - zastąp niszowym twistem architektonicznym');
        }
      }
    }
    if (parsed.metadata) {
      const validLanguages = ['TypeScript/Node.js', 'Python', 'Rust', 'Go', 'C#/.NET', 'Java', 'C++'];
      if (parsed.metadata.language && !validLanguages.includes(parsed.metadata.language)) {
        issues.push('Nieprawidłowy language: "' + parsed.metadata.language + '" - musi być jednym z: ' + validLanguages.join(', '));
      }
      if (typeof parsed.metadata.is_chaotic_user !== 'boolean') {
        warnings.push('⚠ metadata.is_chaotic_user powinno być boolean (true/false)');
      }
      if (typeof parsed.metadata.is_deep_repo !== 'boolean') {
        warnings.push('⚠ metadata.is_deep_repo powinno być boolean (true/false)');
      }
    }
    return { valid: issues.length === 0, issues, warnings };
  }

  function validateAndProcess(rawText) {
    const rawLen = rawText.trim().length;
    if (!rawText.trim().endsWith('}')) {
      LOG('❌ Wykryto urwany JSON (brak zamknięcia)');
      return { status: 'invalid', text: rawText, issues: ['INVALID_INCOMPLETE_JSON – brak końcowego }'] };
    }
    if (rawLen < 2500)
      return { status: 'too_short', text: rawText, issues: ['Za krótki: ' + rawLen + ' zn.'] };
    let parsed = null, fixed = rawText.trim(), wasFixed = false;
    try {
      parsed = JSON.parse(fixed);
    } catch {
      LOG('⚠ JSON.parse failed, auto-fix...');
      fixed = autoFix(rawText);
      try { parsed = JSON.parse(fixed); wasFixed = true; LOG('🔧 Auto-fix OK'); }
      catch (e2) { return { status: 'invalid', text: rawText, issues: ['JSON error: ' + e2.message] }; }
    }
    const { valid, issues, warnings } = validateStructure(parsed);
    if (valid) {
      if (wasFixed) return { status: 'fixed', text: fixed, issues: warnings };
      return { status: 'valid', text: fixed, issues: warnings };
    }
    const critical = issues.filter(i =>
      i.includes('messages') || (i.includes('<think>') && !i.includes('długość')) ||
      i.includes('assistant') || i.includes('Za krótki') || i.includes('Brak pola') ||
      i.includes('Nieznane narzędzie') || i.includes('Brak arg') ||
      i.includes('is_multi_turn:true') || i.includes('wymagany medium+') ||
      i.includes('wymagane medium+') || i.includes('wymagane hard/extreme') ||
      i.includes('Output za krótki') || i.includes('brak sekcji') ||
      i.includes('Priorytet 5') || i.includes('PRIORYTET 6') ||
      i.includes('is_chaotic_user') || i.includes('is_deep_repo') || i.includes('language') ||
      i.includes('ANTI-FLUFF') || i.includes('STRICT_CONSTRAINT_VIOLATION') || i.includes('SYSTEM_FAILURE_VALIDATION') ||
      i.includes('FAKE_API') || i.includes('CROSS_FILE') || i.includes('DEAD_CODE') ||
      i.includes('COMPLEXITY') || i.includes('DEP_VERSION') || i.includes('TOKEN_EFF') ||
      i.includes('ADVERSARIAL') || i.includes('SEMANTIC_DUPLICATE')
    );
    if (critical.length > 0)
      return { status: 'invalid', text: wasFixed ? fixed : rawText, issues: [...critical, ...warnings] };
    if (wasFixed) return { status: 'fixed', text: fixed, issues: [...issues, ...warnings] };
    return { status: 'valid', text: fixed, issues: [...issues, ...warnings] };
  }

  // ========================== SAVE & INTERCEPT (oryginalne) ==========================
  const downloaded = new Set();
  let _clipboardResolve = null;

  function saveFile(result, rawText) {
    if (result.status === 'too_short' || result.status === 'duplicate') return;
    const ts = Date.now();
    let fileContent, filename;
    if (result.status == 'invalid') {
      const errorHeader = result.issues.map(e => '// VALIDATION_ERROR: ' + e).join('\n');
      fileContent = errorHeader + '\n' + rawText + '\n';
      filename = 'REVIEWED_INVALID_' + ts + '.jsonl';
    } else {
      fileContent = result.text + '\n';
      filename = 'REVIEWED_' + ts + '.jsonl';
    }
    const blob = new Blob([fileContent], { type: 'application/jsonl' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.download = filename;
    a.href = url;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
    LOG('💾 Zapisano: ' + filename);
  }

  function downloadJSONL(rawText) {
    if (!rawText || downloaded.has(rawText)) {
      if (_clipboardResolve) {
        const r = _clipboardResolve;
        _clipboardResolve = null;
        r('skipped');
      }
      return;
    }
    downloaded.add(rawText);
    setTimeout(() => downloaded.delete(rawText), 180000);
    const result = validateAndProcess(rawText);
    stats.total++;

    // v69.0 stats tracking
    if (result.issues) {
      if (result.issues.some(i => i.includes('ANTI-FLUFF'))) stats.antiFluffRejected++;
      if (result.issues.some(i => i.includes('STRICT_CONSTRAINT_VIOLATION'))) stats.constraintViolations++;
      if (result.issues.some(i => i.includes('SYSTEM_FAILURE'))) stats.systemFailureDetected++;
    }

    if (result.status == 'too_short') {
      LOG('⚠ ZA KRÓTKI');
      stats.tooShort++; stats.savedInvalid++;
    } else if (result.status == 'valid') {
      LOG('✅ REVIEWED VALID (' + Math.round(result.text.length / 1024) + 'KB)');
      if (result.issues.length) LOG(' ⚠️ ', result.issues.join(' | '));
      stats.saved++;
    } else if (result.status == 'fixed') {
      LOG('🔧 REVIEWED AUTO-FIXED');
      if (result.issues.length) LOG(' ⚠️ ', result.issues.join(' | '));
      stats.saved++; stats.autoFixed++;
    } else {
      if (isHopeless(result.issues)) {
        LOG('💀 PRZYKŁAD BEZNADZIEJNY – odrzucony bez naprawy (' + result.issues.length + ' krytycznych błędów)');
        stats.hopelessRejected++;
        stats.savedInvalid++;
      } else {
        LOG('❌ REVIEWED INVALID');
        result.issues.forEach(i => LOG(' ❌ ' + i));
        stats.savedInvalid++;
        saveFile(result, rawText);
      }
    }
    stats.save();
    if (_clipboardResolve) {
      const r = _clipboardResolve;
      _clipboardResolve = null;
      r(result.status);
    }
    stats.print();
  }

  function installClipboardInterceptor() {
    if (navigator.clipboard._intercepted) return;
    navigator.clipboard._intercepted = true;
    const original = navigator.clipboard.writeText.bind(navigator.clipboard);
    navigator.clipboard.writeText = async (text) => {
      LOG('📦 Przechwycono - walidacja v70.0...');
      downloadJSONL(text);
      return original(text);
    };
    LOG('Interceptor v70.0 aktywny.');
  }

    });
  }

  function simulateClick(el) {
    if (!el) return;
    ['mouseover', 'mousedown', 'mouseup', 'click'].forEach(t => el.dispatchEvent(new MouseEvent(t, { bubbles: true, cancelable: true })));
  }

  function setTextareaValue(ta, text) {
    if (!ta) return;
    ta.focus();
    const setter = Object.getOwnPropertyDescriptor(HTMLTextAreaElement.prototype, 'value').set;
    setter.call(ta, text);
    ta.dispatchEvent(new Event('input', { bubbles: true }));
    ta.dispatchEvent(new Event('change', { bubbles: true }));
  }

  function findCopyButton(msg) {
    for (const btn of msg.querySelectorAll('button, [role="button"]')) {
      if (isInsideThinking(btn)) continue;
      if (btn.getAttribute('aria-disabled') === 'true') continue;
      const txt = (btn.innerText || btn.textContent || '').trim();
      if (txt === 'Kopiuj' || txt === 'Copy') return btn;
    }
    for (const btn of msg.querySelectorAll('button, [role="button"]')) {
      if (isInsideThinking(btn)) continue;
      if (btn.getAttribute('aria-disabled') === 'true') continue;
      const label = (btn.getAttribute('aria-label') || '').toLowerCase();
      if (label.includes('kopi') || label.includes('copy')) return btn;
    }
    const containers = msg.querySelectorAll('[class*="action"], [class*="toolbar"], [class*="footer"], [class*="operate"]');
    for (const c of containers) {
      if (isInsideThinking(c)) continue;
      const btn = c.querySelectorAll('.ds-icon-button[role="button"]:not([aria-disabled="true"])');
      if (btn.length > 0) return btn[0];
    }
    return null;
  }
