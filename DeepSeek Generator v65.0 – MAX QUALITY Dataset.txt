    function validateToolCalls(content, issues) {
        const reNew = /<tool_call\s+name="([^"]+)">\s*<args>(.*?)<\/args>\s*<\/tool_call>/gs;
        let m;
        while ((m = reNew.exec(content)) !== null) {
            const name = m[1];
            let args = {};
            try {
                args = JSON.parse(m[2]);
            } catch { issues.push(`Niepoprawny JSON w <args> dla narzędzia: ${name}`); continue; }
            if (!TOOL_SCHEMAS[name]) { issues.push(`Nieznane narzędzie: "${name}"`); continue; }
            const schema = TOOL_SCHEMAS[name];
            schema.required.forEach(r => { if (!(r in args)) issues.push(`Brak arg "${r}" w ${name}()`); });
            const allowed = [...schema.required, ...schema.optional];
            Object.keys(args).forEach(k => { if (!allowed.includes(k)) issues.push(`Nieznany arg "${k}" w ${name}()`); });
        }
        const reOld = /<tool_call>(\{.*?\})<\/tool_call>/gs;
        while ((m = reOld.exec(content)) !== null) {
            try {
                const tc = JSON.parse(m[1]);
                const name = tc.name;
                const args = tc.args || {};
                issues.push(`Stary format tool_call bez atrybutu name wykryty dla: "${name}" – użyj <tool_call name="${name}"><args>...</args></tool_call>`);
                if (!TOOL_SCHEMAS[name]) { issues.push(`Nieznane narzędzie: "${name}"`); continue; }
                const schema = TOOL_SCHEMAS[name];
                schema.required.forEach(r => { if (!(r in args)) issues.push(`Brak arg "${r}" w ${name}()`); });
            } catch { issues.push('Niepoprawny JSON w starym formacie <tool_call>'); }
        }
    }

    function fixNewlinesInJsonStrings(str) {
        let result = '', inString = false, escaped = false;
        for (let i = 0; i < str.length; i++) {
            const ch = str[i];
            if (escaped) { result += ch; escaped = false; continue; }
            if (ch === '\\' && inString) { result += ch; escaped = true; continue; }
            if (ch === '"') { inString = !inString; result += ch; continue; }
            if (inString) {
                if (ch === '\n') { result += '\\n'; continue; }
                if (ch === '\r') { result += '\\r'; continue; }
                if (ch === '\t') { result += '\\t'; continue; }
                const code = ch.charCodeAt(0);
                if (code < 0x20) { result += '\\u' + code.toString(16).padStart(4, '0'); continue; }
            }
            result += ch;
        }
        return result;
    }

    function autoFix(raw) {
        let text = raw.trim();
        text = text.replace(/^\uFEFF/, '').replace(/\0/g, '');
        text = text.replace(/^```(?:json)?\s*/i, '').replace(/\s*```\s*$/, '').trim();
        const f = text.indexOf('{'), l = text.lastIndexOf('}');
        if (f !== -1 && l !== -1 && l > f && (f > 0 || l < text.length - 1))
            text = text.slice(f, l + 1);
        text = text.replace(/,\s*([}\]])/g, '$1');
        text = fixNewlinesInJsonStrings(text);
        text = text.replace(/(?<!\\)\\(?!["\\/bfnrtu])/g, '\\\\');
        return text;
    }

    function estimateTokens(text) {

    }

    // 2. Cross-File Consistency (spójność między plikami)
    function checkCrossFileConsistency(content, issues) {
        const fileSections = content.split(/\/\/ FILE:\s*(.+)/);
        const files = {};
        for (let i = 1; i < fileSections.length; i += 2) {
            const filePath = fileSections[i].trim();
            const fileContent = fileSections[i + 1] || '';
            files[filePath] = fileContent;
        }

        const definedSymbols = {};
        for (const [path, code] of Object.entries(files)) {
            const defs = [];
            const funcRegex = /(?:function|def|func|fn|class|const|let|var)\s+(\w+)/g;
            let m;
            while ((m = funcRegex.exec(code)) !== null) defs.push(m[1]);
            definedSymbols[path] = defs;
        }

        for (const [path, code] of Object.entries(files)) {
            const importRegex = /(?:import|require|from)\s+(?:\{?\s*(\w+)\s*\}?\s*from\s+['"]\.?\/?([^'"]+)['"]|['"](\w+)['"])/g;
            let m;
            while ((m = importRegex.exec(code)) !== null) {
                const importedName = m[1] || m[3];
                const targetFile = m[2];
                if (!importedName) continue;
                let found = false;
                for (const [fpath, symbols] of Object.entries(definedSymbols)) {
                    if (fpath.includes(targetFile)) {
                        if (symbols.includes(importedName)) found = true;
                        break;
                    }
                }
                if (!found) {
                    issues.push(`CROSS_FILE: Import "${importedName}" w pliku ${path} nie został znaleziony w pliku docelowym.`);
                    stats.crossFileInconsistencies++;
                }
            }
        }
    }

    // 3. Dead Code Detection
    function detectDeadCode(content, issues) {
        const funcDefs = content.match(/(?:function|def|fn|func|class|const|let|var)\s+(\w+)/g) || [];
        const usedSymbols = new Set();
        const usageRegex = /\b(\w+)\s*\(/g;
        let m;
        while ((m = usageRegex.exec(content)) !== null) usedSymbols.add(m[1]);
        for (const def of funcDefs) {
            const name = def.split(/\s+/)[1];
            if (!usedSymbols.has(name) && !['main', 'index', 'handler', 'init'].includes(name)) {
                issues.push(`DEAD_CODE: Zdefiniowano "${def}" ale nigdzie nie użyto.`);
                stats.deadCodeOccurrences++;
            }
        }
    }

    // 4. Semantic Duplicate Detection (prosty fingerprint)
    const recentFingerprints = [];
    function detectSemanticDuplicate(parsed) {
        if (!parsed.metadata) return;
        const fingerprint = `${parsed.metadata.language}|${parsed.metadata.domain}|${parsed.metadata.project_type}|${parsed.metadata.unique_twist}`;
        if (recentFingerprints.includes(fingerprint)) {
            stats.semanticDuplicates++;
            return true;
        }
        recentFingerprints.push(fingerprint);
        if (recentFingerprints.length > 20) recentFingerprints.shift();
        return false;
    }

    // 5. Cyclomatic Complexity Gate
    function checkCyclomaticComplexity(content, difficulty, issues) {
        const complexity = (content.match(/\b(if|else\s+if|for|while|switch|case\b|catch)\b/g) || []).length + 1;
        const limits = { easy: [1, 4], medium: [5, 12], hard: [13, 25], extreme: [26, 100] };
        const [min, max] = limits[difficulty] || [0, Infinity];
        if (complexity < min || complexity > max) {
            issues.push(`COMPLEXITY: Złożoność cyklomatyczna ${complexity} poza zakresem dla ${difficulty} (${min}-${max}).`);
            stats.complexityViolations++;
        }
    }

    // 6. Dependency Version Consistency
    function checkDependencyVersions(content, issues) {
        const versionRegex = /"version"\s*:\s*"(\d+\.\d+\.\d+)"/g;
        let m;
        while ((m = versionRegex.exec(content)) !== null) {
            const ver = m[1];
            if (!/^\d+\.\d+\.\d+$/.test(ver)) {
                issues.push(`DEP_VERSION: Niewłaściwy format wersji "${ver}".`);
                stats.dependencyVersionIssues++;
            }
        }
    }

    // 7. Token Efficiency Score
    function checkTokenEfficiency(content, issues) {
        const operationalCount = (content.match(/(<tool_call|app\.(get|post|put|delete)|db\.(query|execute)|\.test\(|\.assert)/g) || []).length;
        const tokens = estimateTokens(content);
        if (tokens > 0 && operationalCount / tokens < 0.02) {
            issues.push(`TOKEN_EFF: Niska gęstość operacji (${(operationalCount/tokens*100).toFixed(1)}%) – kod może być rozwlekły.`);
            stats.tokenEfficiencyWarnings++;
        }
    }

    // 8. Adversarial Self-Test
    function checkAdversarialTests(content, issues) {
        const testSection = content.match(/(?:describe|it|test|def test_|fn test_)/g) || [];
        const adversarialIndicators = /(?:empty|null|undefined|NaN|extreme|boundary|invalid|negative|zero|large|overflow|injection|escape|script)/i;
        let hasAdversarial = false;
        for (const test of testSection) {
            if (adversarialIndicators.test(test)) {
                hasAdversarial = true;
                break;
            }
        }
        if (!hasAdversarial && testSection.length > 0) {
            issues.push(`ADVERSARIAL: Brak testów przypadków brzegowych (np. puste wejście, ekstremalne wartości).`);
            stats.adversarialTestMissing++;
        }
    }

    function validateStructure(parsed, meta) {
        const issues = [], warnings = [];
        const difficulty = parsed?.metadata?.difficulty || 'unknown';

        if (!parsed.messages || !Array.isArray(parsed.messages)) {
            issues.push('Brak pola "messages"');
            return { valid: false, issues, warnings };
        }
        if (parsed.messages.length !== 3)
            issues.push(`messages ma ${parsed.messages.length} el. zamiast 3`);

        ['system', 'user', 'assistant'].forEach((role, i) => {
            if (parsed.messages[i]?.role !== role)
                issues.push(`messages[${i}].role ≠ "${role}"`);
            if (!parsed.messages[i]?.content || parsed.messages[i].content.length < 10)
                issues.push(`messages[${i}].content za krótki`);
        });

        const am = parsed.messages.find(m => m.role === 'assistant');
        if (am?.content) {
            const c = am.content;
            if (!c.includes('<think>')) issues.push('Brak <think>');
            if (!c.includes('</think>')) issues.push('Brak </think>');

            const te = c.indexOf('</think>');
            const ts = c.indexOf('<think>');
            if (ts !== -1 && te !== -1 && te > ts) {
                const thinkContent = c.slice(ts + 7, te);
                const thinkTokens = estimateTokens(thinkContent);
                if (thinkTokens < 400) issues.push(`<think> za krótki: ~${thinkTokens} tokenów (min 400)`);

                if (!thinkContent.includes('## 1. Architektura i Ograniczenia'))
                    issues.push('<think> brak sekcji "## 1. Architektura i Ograniczenia"');
                if (!thinkContent.includes('## 2. Threat Modeling'))
                    issues.push('<think> brak sekcji "## 2. Threat Modeling"');
                if (!thinkContent.includes('## 3. Strategia ReAct'))
                    issues.push('<think> brak sekcji "## 3. Strategia ReAct"');
                if (!thinkContent.includes('## 4. JSONL Self-Check'))
                    issues.push('<think> brak sekcji "## 4. JSONL Self-Check"');
            }

            if (te !== -1) {
                const after = c.slice(te + 8).trim();
                if (after.length < 150) issues.push('Output po </think> za krótki');
                if (!(after.match(/\/\/ FILE:/g) || []).length) issues.push('Brak // FILE:');
                if (!after.includes('<tool_call')) issues.push('Brak <tool_call>');

                const afterTokens = estimateTokens(after);
                const minTokens = { easy: 150, medium: 400, hard: 800, extreme: 1400 };
                if (minTokens[difficulty] && afterTokens < minTokens[difficulty]) {
                    issues.push(`Output za krótki dla ${difficulty}: ~${afterTokens} tokenów (min ${minTokens[difficulty]})`);
                }

                if (!after.includes('git_commit')) {
                    if (difficulty === 'easy') warnings.push('Brak git_commit (opcjonalny easy)');
                    else if (difficulty !== 'unknown') issues.push('Brak git_commit (wymagany medium+)');
                    else warnings.push('Brak git_commit');
                }
                if (!after.includes('run_tests')) {
                    if (difficulty === 'easy') warnings.push('Brak run_tests (opcjonalne easy)');
                    else if (difficulty !== 'unknown') issues.push('Brak run_tests (wymagane medium+)');
                    else warnings.push('Brak run_tests');
                }
                if (!after.includes('run_security_scan') && (difficulty === 'hard' || difficulty === 'extreme'))
                    issues.push('Brak run_security_scan (wymagane hard/extreme)');
                if (!after.includes('BENCHMARK') && (difficulty === 'hard' || difficulty === 'extreme'))
                    warnings.push('Brak BENCHMARK (zalecane hard/extreme)');
                if (!after.includes('retry') && !after.includes('fallback') &&
                    !after.includes('circuit') && !after.includes('SELF-CORRECTION'))
                    warnings.push('Brak error recovery pattern');
                if (!after.includes('zod') && !after.includes('pydantic') &&
                    !after.includes('class-validator') && !after.includes('validate'))
                    warnings.push('Brak walidacji inputów');

                validateToolCalls(after, issues);

                // WYWOŁANIE NOWYCH WALIDATORÓW v70.0
                detectFakeApiCalls(after, parsed.metadata?.language || 'unknown', issues);
                checkCrossFileConsistency(after, issues);
                detectDeadCode(after, issues);
                checkCyclomaticComplexity(after, difficulty, issues);
                checkDependencyVersions(after, issues);
                checkTokenEfficiency(after, issues);
                checkAdversarialTests(after, issues);

                if (detectSemanticDuplicate(parsed)) {
                    issues.push('SEMANTIC_DUPLICATE: Przykład ma identyczny fingerprint jak jeden z ostatnich 20.');
                }

                // chaos/deep_repo checks
                if (meta && meta.is_chaotic_user) {
                    const beforeFirstWrite = c.substring(0, c.indexOf('<tool_call name="write_file"'));
                    const beforeFirstReplace = c.substring(0, c.indexOf('<tool_call name="replace_in_file"'));
                    const beforeModification = beforeFirstWrite.length > 0 ? beforeFirstWrite : beforeFirstReplace;
                    if (beforeModification.length === 0 || !beforeModification.includes('ask_clarification')) {
                        issues.push('is_chaotic_user:true – brak ask_clarification przed modyfikacjami plików!');
                    }
                }
                if (meta && meta.is_deep_repo) {
                    if (!c.includes('<tool_call name="list_files"')) {
                        issues.push('is_deep_repo:true – brak list_files!');
                    }
                    if (!c.includes('<tool_call name="search_code"')) {
                        issues.push('is_deep_repo:true – brak search_code!');
                    }
                    if (!c.includes('<tool_call name="read_file"')) {
                        issues.push('is_deep_repo:true – brak read_file!');
                    }
                }
            }
        }

        if (!parsed.metadata) {
            issues.push('Brak metadata');
        } else {
            ['category', 'difficulty', 'project_type', 'instruction_format',
                'is_negative_example', 'is_multi_turn', 'unique_id', 'domain']
                .forEach(k => { if (!(k in parsed.metadata)) issues.push(`metadata.${k} brakuje`); });
        }

        return { valid: issues.length === 0, issues, warnings };
    }

    function validateAndProcess(rawText) {
        stats.total++;
        const rawLen = rawText.trim().length;

        if (rawLen < 2500) {
            stats.tooShort++; stats.invalid++;
            return { status: 'too_short', text: rawText, issues: [`Za krótki: ${rawLen} zn.`] };
        }

        const contentHash = simpleHash(rawText);
        if (sessionHashes.has(contentHash)) {
            stats.duplicates++; stats.invalid++;
            LOG('🔁 DUPLIKAT wykryty przez hash – pomijam.');
            return { status: 'duplicate', text: rawText, issues: ['Duplikat zawartości'] };
        }

        let parsed = null, fixed = rawText.trim(), wasFixed = false;
        try {
            parsed = JSON.parse(fixed);
        } catch {
            LOG('⚠ JSON parse failed, próba auto-fix...');
            fixed = autoFix(rawText);
            try { parsed = JSON.parse(fixed); wasFixed = true; LOG('🔧 Auto-fix OK'); }
            catch (e2) { stats.invalid++; return { status: 'invalid', text: rawText, issues: [`JSON error: ${e2.message}`] }; }
        }

        const { valid, issues, warnings } = validateStructure(parsed, currentCycleMeta);

        if (parsed.metadata) {
            updateDomainTracking(parsed);
            const uid = parsed.metadata.unique_id;
            if (uid && seenUniqueIds.has(uid)) {
                stats.duplicates++;
                warnings.push(`⚠ DUPLIKAT unique_id: ${uid}`);
            } else if (uid) {
                seenUniqueIds.add(uid);
            }

            const d = parsed.metadata.difficulty || 'unknown';
            stats.byDifficulty[d] = (stats.byDifficulty[d] || 0) + 1;
            const ts2 = parsed.metadata.teaching_style || 'unknown';
            stats.byTeachingStyle[ts2] = (stats.byTeachingStyle[ts2] || 0) + 1;
            const spv = parsed.metadata.system_prompt_variant;
            if (spv && [1, 2, 3, 4].includes(spv)) stats.bySystemPromptVariant[spv]++;
            else stats.bySystemPromptVariant.unknown++;
            if (parsed.metadata.is_multi_turn === true) stats.multiTurn++;
            if (parsed.metadata.is_negative_example === true) stats.negativeExamples++;
            const lang = parsed.metadata.language || 'unknown';
            stats.byLanguage[lang] = (stats.byLanguage[lang] || 0) + 1;
            if (parsed.metadata.is_chaotic_user === true) stats.chaoticUser++;
            if (parsed.metadata.is_deep_repo === true) stats.deepRepo++;
            const domain = parsed.metadata.domain || 'unknown';
            stats.byDomain[domain] = (stats.byDomain[domain] || 0) + 1;
            const ac = parsed.messages?.find(m => m.role === 'assistant')?.content || '';
            if (ac.includes('SELF-CORRECTION')) stats.selfCorrection++;
            lastGeneratedText = ac;
            const rotWarnings = warnings.filter(w => w.includes('ROTACJA'));
            if (rotWarnings.length > 0) stats.poorRotation++;
            if (parsed.metadata.teaching_style) lastTeachingStyle = parsed.metadata.teaching_style;
            if (parsed.metadata.system_prompt_variant) lastSystemPromptVariant = parsed.metadata.system_prompt_variant;
            if (parsed.metadata.difficulty) lastDifficulty = parsed.metadata.difficulty;
        } else {
            stats.byDifficulty.unknown++;
            stats.byTeachingStyle.unknown++;
            stats.bySystemPromptVariant.unknown++;
            stats.byLanguage.unknown++;
        }

        if (valid) {
            if (warnings.length > 0) stats.warningsOnly++;
            sessionHashes.add(contentHash);
            saveStats();
            if (wasFixed) { stats.autoFixed++; stats.valid++; return { status: 'fixed', text: fixed, issues: warnings }; }
            stats.valid++;
            return { status: 'valid', text: fixed, issues: warnings };
        }

        const critical = issues.filter(i =>
            i.includes('messages') || (i.includes('<think>') && !i.includes('długość')) ||
            i.includes('assistant') || i.includes('Za krótki') || i.includes('Brak pola') ||
            i.includes('Nieznane narzędzie') || i.includes('Brak arg') ||
            i.includes('is_multi_turn:true') || i.includes('wymagany medium+') ||
            i.includes('wymagane medium+') || i.includes('wymagane hard/extreme') ||
            i.includes('Output za krótki') || i.includes('brak sekcji') ||
            i.includes('is_chaotic_user') || i.includes('is_deep_repo') ||
            i.includes('FAKE_API') || i.includes('CROSS_FILE') || i.includes('DEAD_CODE') ||
            i.includes('COMPLEXITY') || i.includes('DEP_VERSION') || i.includes('TOKEN_EFF') ||
            i.includes('ADVERSARIAL') || i.includes('SEMANTIC_DUPLICATE')
        );

        if (critical.length > 0) {
            stats.invalid++;
            return { status: 'invalid', text: fixed, issues: [...critical, ...warnings] };
        }

        sessionHashes.add(contentHash);
        if (wasFixed) stats.autoFixed++;
        stats.valid++; stats.warningsOnly++;
        saveStats();
        return { status: wasFixed ? 'fixed' : 'valid', text: fixed, issues: [...issues, ...warnings] };
    }

    const downloaded = new Set();
    let _resolveClipboard = null;

    function saveFile(result, rawText) {
        if (result.status === 'too_short' || result.status === 'duplicate') return;
        const ts = Date.now();
        let fileContent, filename;

        if (result.status === 'invalid') {
            const errorHeader = result.issues.map(e => `// VALIDATION_ERROR: ${e}`).join('\n');
            fileContent = errorHeader + '\n' + rawText + '\n';
            filename = `INVALID_${ts}.jsonl`;
        } else {
            fileContent = result.text + '\n';
            filename = `deepseek_export_${ts}.jsonl`;
        }

        const blob = new Blob([fileContent], { type: 'application/jsonl' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.download = filename;
        a.href = url;
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
        URL.revokeObjectURL(url);
        LOG(`📁 Zapisano: ${filename}`);
    }

    function downloadJSONL(rawText) {
        if (!rawText || downloaded.has(rawText)) {
            if (_resolveClipboard) {
                const resolve = _resolveClipboard;
                _resolveClipboard = null;
                resolve('skipped');
            }
            return;
        }
        downloaded.add(rawText);
        setTimeout(() => downloaded.delete(rawText), 180000);

        const result = validateAndProcess(rawText);

        if (result.status === 'too_short') LOG(`📏 ZA KRÓTKI – pomijam.`);
        else if (result.status === 'duplicate') LOG(`🔁 DUPLIKAT – pomijam.`);
        else if (result.status === 'valid') LOG(`✅ VALID`);
        else if (result.status === 'fixed') LOG(`🔧 AUTO-FIXED`);
        else {
            LOG(`❌ INVALID`);
            result.issues.forEach(i => LOG(`  ⚠ ${i}`));
        }

        saveFile(result, rawText);

        if (_resolveClipboard) {
            const resolve = _resolveClipboard;
            _resolveClipboard = null;
            resolve(result.status);
        }

        if (stats.total % 10 === 0) stats.printSummary();
    }

    function installClipboardInterceptor() {
        if (navigator.clipboard.__intercepted) return;
        navigator.clipboard.__intercepted = true;
        const original = navigator.clipboard.writeText.bind(navigator.clipboard);
        navigator.clipboard.writeText = async (text) => {
            LOG(`🔌 Interceptor: ${text.length} znaków`);
            downloadJSONL(text);
            return original(text);
        };
        LOG('🔌 Interceptor schowka aktywny.');
    }

    function waitForClipboardCapture(timeoutMs = 12000) {
        return new Promise((resolve) => {
            _resolveClipboard = resolve;
            setTimeout(() => {
                if (_resolveClipboard) {
                    LOG('⚠ Timeout interceptora');
                    _resolveClipboard = null;
                    resolve(null);
                }
            }, timeoutMs);
        });
    }

    function simulateClick(el) {
        if (!el) return;
        ['mouseover', 'mousedown', 'mouseup', 'click'].forEach(t =>
            el.dispatchEvent(new MouseEvent(t, { bubbles: true, cancelable: true }))
        );
    }

    function setTextareaValue(ta, text) {
        ta.focus();
        const s = Object.getOwnPropertyDescriptor(HTMLTextAreaElement.prototype, 'value').set;
        s.call(ta, text);
        ta.dispatchEvent(new Event('input', { bubbles: true }));
        ta.dispatchEvent(new Event('change', { bubbles: true }));
    }

    function isActiveSendArrow() {
        const btn = getSendButton();

        if (btn.classList.contains('ds-icon-button--disabled')) return false;
        if (btn.getAttribute('aria-disabled') === 'true') return false;
        const d = btn.querySelector('svg path')?.getAttribute('d') || '';
        return d.includes('M8.3125') && !d.includes('M4.88');
    }

    function isInactiveSendArrow() {
        const btn = getSendButton();
        if (!btn) return false;
        if (!btn.classList.contains('ds-icon-button--disabled') && btn.getAttribute('aria-disabled') !== 'true') return false;
        const d = btn.querySelector('svg path')?.getAttribute('d') || '';
        return d.includes('M8.3125') && !d.includes('M4.88');
    }

    async function waitForActiveSendButton(ms = 15000) {
        const t = Date.now();
        while (Date.now() - t < ms) {
            if (isActiveSendArrow()) return getSendButton();
            await wait(200);
        }

    }

    async function waitForInactiveArrow() {
        const t = Date.now();
        let last = Date.now();
        while (true) {
            if (isInactiveSendArrow()) {
                LOG(`✅ Gotowe po ${Math.round((Date.now() - t) / 1000)}s`);
                return true;
            }
            if (Date.now() - last > 60000) {
                LOG(`⏳ ${Math.round((Date.now() - t) / 1000)}s... nadal generuje, czekam dalej bez limitu`);
                last = Date.now();
            }
            await wait(400);
        }
    }

    function tryCopyMethod1(msg) {
        for (const btn of msg.querySelectorAll('button.ds-atom-button.ds-text-button--with-icon')) {
            if (isInsideThinking(btn) || btn.getAttribute('aria-disabled') === 'true') continue;

        return false;
    }

    function tryCopyMethod2(msg) {
        for (const c of msg.querySelectorAll('[class*="_965abe9"]')) {
            if (isInsideThinking(c)) continue;
            const ib = c.querySelector('.db183363.ds-icon-button');
            if (ib && ib.getAttribute('aria-disabled') !== 'true') {
                LOG('✅ [M2] Klikam .db183363.ds-icon-button');
                simulateClick(ib);
                return true;
            }
        }
        return false;
    }

    function tryCopyMethod3(lastMsg) {
        const actionBars = lastMsg.querySelectorAll('.ds-flex');
        for (const bar of actionBars) {
            if (isInsideThinking(bar)) continue;
            const icons = bar.querySelectorAll('.ds-icon-button[role="button"]');
            if (icons.length >= 2) {
                const firstIcon = icons[0];
                if (firstIcon.getAttribute('aria-disabled') !== 'true') {
                    LOG('✅ [M3] Klikam pierwszy icon-button w pasku akcji');
                    simulateClick(firstIcon);
                    return true;
                }
            }
        }
        const allIcons = lastMsg.querySelectorAll('.ds-icon-button[role="button"]');
        for (const icon of allIcons) {
            if (isInsideThinking(icon)) continue;
            if (icon.getAttribute('aria-disabled') === 'true') continue;
            LOG('⚠ [M3 fallback] Klikam pierwszy aktywny icon-button');
            simulateClick(icon);
            return true;
        }
        return false;
    }

    async function tryCopyMethod4(msg) {
        const mc = msg.querySelector('.ds-assistant-message-main-content');
        const text = mc ? mc.innerText.trim() : (() => {
            const cl = msg.cloneNode(true);
            cl.querySelectorAll('.ds-think-content, .e1675d8b').forEach(e => e.remove());
            return cl.innerText.trim();
        })();
        if (!text) return false;
        try {
            LOG('✅ [M4] Bezpośredni zapis do clipboard');
            await navigator.clipboard.writeText(text);
            return true;
        } catch {
            return false;
        }
    }

    async function hybridCopy() {
        await scrollToBottomAndWait();
        const lastMsg = getLastAssistantMessage();
        if (!lastMsg) {
            LOG('❌ Brak wiadomości asystenta w DOM');
            return false;
        }
        if (tryCopyMethod1(lastMsg)) return true;
        LOG('M1 nieudana → M2...');
        if (tryCopyMethod2(lastMsg)) return true;
        LOG('M2 nieudana → M3...');
        if (tryCopyMethod3(lastMsg)) return true;
        LOG('M3 nieudana → M4...');
        return await tryCopyMethod4(lastMsg);
    }

    async function copyWithRetry(ms = 3000) {
        let i = 0;
        while (true) {
            i++;
            LOG(`📋 Próba kopiowania ${i}`);
            const capturePromise = waitForClipboardCapture(12000);
            const ok = await hybridCopy();
            if (ok) {
                const status = await capturePromise;
                if (status !== null) {
                    LOG(`✅ Copy przechwycone (${status})`);
                } else {
                    LOG(`⚠ Copy wykonane, ale nie przechwycone – kontynuuję cykl.`);
                }
                return true;
            }
            if (i % 20 === 0) {
                LOG(`⚠ ${i} prób kopiowania – kontynuuję bez limitu...`);
            }
            await wait(ms);
        }
    }
